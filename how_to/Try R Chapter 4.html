Mean

names(limbs) <- c("One-Eye", "Peg-Leg", "Smitty", "Hook", "Scooter", "Dan",

Median

    "Mikey", "Davy Jones")

mean(limbs)

## [1] 4.75

abline(h = mean(limbs))

## [1] 4

Пример

Standard Deviation

# ---
pounds <- c(45000, 50000, 35000, 40000, 35000, 45000, 10000, 15000)
meanValue <- mean(pounds)
# ---
deviation <- sd(pounds)
deviation

## [1] 14501

# ---
barplot(pounds)
abline(h = meanValue)
abline(h = meanValue + deviation)
abline(h = meanValue - deviation)
