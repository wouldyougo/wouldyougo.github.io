Пример

Vector Access

sentence <- c("walk", "the", "plank")
sentence

## [1] "walk"  "the"   "plank"


sentence[3]

## [1] "plank"

sentence[1]

## [1] "walk"

sentence[3] <- "dog"
sentence[4] <- "to"
sentence

## [1] "walk" "the"  "dog"  "to"


sentence[c(1, 3)]

## [1] "walk" "dog"

sentence[2:4]

## [1] "the" "dog" "to"

sentence[5:7] <- c("the", "poop", "deck")
sentence

## [1] "walk" "the"  "dog"  "to"   "the"  "poop" "deck"
