## Loading required package: car


# Single scatterplot with groups marked Function can be called 'scatterplot'
# or 'sp'
sp(Sepal.Width ~ Sepal.Length | Species, data = iris, xlab = "Sepal Width", 
    ylab = "Sepal Length", main = "Iris Data", labels = row.names(iris))
